app.post("/webhook", express.raw({ type: "application/json" }), async (req, res) => {
  let event;
  try {
    event = stripe.webhooks.constructEvent(
      req.body,
      req.headers["stripe-signature"],
      process.env.STRIPE_WEBHOOK_SECRET
    );
  } catch (err) {
    return res.status(400).send(`Webhook Error: ${err.message}`);
  }

  console.log("ðŸ“© Event:", event.type);

  switch (event.type) {
    case "checkout.session.completed":
      const session = event.data.object;
      await handleDonationSuccess(session);
      break;

    case "payment_intent.payment_failed":
      const piFail = event.data.object;
      await handleDonationFailed(piFail);
      break;

    case "charge.refunded":
      const refund = event.data.object;
      await handleDonationRefund(refund);
      break;

    case "charge.dispute.created":
      const dispute = event.data.object;
      await handleDonationDispute(dispute);
      break;

    case "checkout.session.expired":
      const expired = event.data.object;
      await handleSessionExpired(expired);
      break;

    default:
      console.log(`Unhandled event type ${event.type}`);
  }

  res.status(200).send("Received");
});
